SELECT *
FROM (
    SELECT receipt_date, team_name, account_number
    FROM your_table
    WHERE receipt_date BETWEEN SYSDATE AND SYSDATE + 15
)
PIVOT (
    COUNT(account_number)
    FOR team_name IN ('Team A', 'Team B', 'Team C', 'Team D', 'Team E', 'Team F', 'Team G', 'Team H')
)
ORDER BY receipt_date;
