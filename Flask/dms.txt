def column_letter_to_index(column_letter):
    column_index = 0
    for letter in column_letter:
        column_index = column_index * 26 + (ord(letter.upper()) - ord('A')) + 1
    return column_index

# Example usage:
column_letter = "AZ"
column_index = column_letter_to_index(column_letter)
print(f"The column index of {column_letter} is {column_index}")
